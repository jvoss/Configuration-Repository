228:   def self.parse_host_string(host_string, options)
229:     
230:     filename = nil
231:     hostname = nil
232:     username = options[:username]
233:     password = options[:password]
234:     
235:     if host_string.include?('file:')
236:       
237:       filename = host_string.split('file:')[1]
238:       
239:     end # if host.string.include('file:')
240:     
241:     if host_string.include?('@')
242:       userpass, hostname = host_string.split(/(.*)@(.*)$/)[1..2]
243:       username, password = userpass.split(/^(\w+):(.*)/)[1..2]
244:       
245:       # userpass split fails when no password is supplied but a user is
246:       # example: user@host.domain.tld
247:       # this will resplit userpass in this condition to take the username only
248:       username = userpass.split(/^(\w+):(.*)/)[0] if username.nil?
249:     else
250:       hostname = host_string
251:     end # host_string.include?('@')
252:     
253:     return filename ? filename : [hostname, username, password]
254:
